二叉树四种遍历的非递归方式

发表于 2021-12-02 更新于 2021-12-07 分类于 数据结构

一、前序、中序和后续遍历

需用到栈：

typedef struct StackNode
{
    BiTree data;
    struct StackNode *next;
} StackNode, *LinkStackPtr;

typedef struct
{
    LinkStackPtr top;
    int count;
} LinkStack;

Status InitStack(LinkStack *S)
{
    S->top = NULL;
    S->count = 0;
    return OK;
}
Status Push(LinkStack *S, BiTree e)
{
    if(NULL == e)
        return ERROR;
    LinkStackPtr s = (LinkStackPtr)malloc(sizeof(StackNode));   //s指向新栈结点的指针
    s->data = e;
    s->next = S->top; /* 把当前的栈顶元素赋值给新结点的直接后继 */
    S->top = s;       /* 将新的结点s赋值给栈顶指针 */
    S->count++;
    return OK;
}
Status Pop(LinkStack *S, BiTree *e)
{
    LinkStackPtr p;
    if ((*S).count == 0)
        return ERROR;
    *e = S->top->data;
    p = S->top;            /* 将栈顶结点赋值给p */
    S->top = S->top->next; /* 使得栈顶指针下移一位，指向后一结点 */
    free(p);               /* 释放结点p */
    S->count--;
    return OK;
}
Status GetTop(LinkStack S, BiTree &p)
{
    if (S.top == NULL)
        return ERROR;
    else
        p = S.top->data;
    return OK;
}

void PreOrderTraverse_2(BiTree T)
{
    LinkStack S;
    InitStack(&S);
    BiTree p = T;
    while(p || S.count != 0)
    {
        if(p)        //?  左子树不为空
        {
            printf("%c", p->data);
            Push(&S, p);    //?  栈存放各子树的右子树
            p = p->lchild;
        }
        else
        {
            Pop(&S, &p);
            p = p->rchild;
        }
    }//?  当左子树为空，且栈空，退出循环
    return;
}

void InOrderTraverse_2(BiTree T)
{
    LinkStack S;
    InitStack(&S);
    BiTree p = T;
    while(p || S.count != 0)
    {
        if(p)        //?  左子树不为空
        {
            Push(&S, p);    //?  栈存放各子树的根结点
            p = p->lchild;
        }
        //?  左子树空，弹出栈顶元素并输出，转为右子树继续循环

        {
            Pop(&S, &p);
            printf("%c",p->data);
            p = p->rchild;
        }

    return;
}

后序遍历实现：

//*  初始条件：二叉树T存在
//todo  操作结果：后序非递归遍历T
void PostOrderTraverse_2(BiTree T)
{
    BiTree p = T, r = NULL;
    LinkStack S;
    InitStack(&S);
    while(p || S.count != 0)
    {
        if(p)        //todo  走到最左边
        {
            Push(&S, p);
            p = p->lchild;
        }
        else
        {
            GetTop(S, p);
            if(p->rchild && p->rchild != r)        //*  若右子树存在, 且未被访问过
                p = p->rchild;        //todo 转向右边
            else
            {
                Pop(&S, &p);
                visit(p->data);
                r = p;        //*  记录最近访问过的结点
                p = NULL;    //?  将p置空
            }
        }//else
    }//while
    return;
}

二、层次遍历

需用到队列：

typedef struct
{
    BiTree *data;
    int front; /* 头指针 */
    int rear;  /* 尾指针，若队列不空，指向队列尾元素的下一个位置 */
} SqQueue;

Status InitQueue(SqQueue &Q)
{
    Q.data = (BiTree *)malloc((MAXSIZE+1) * sizeof(BiTree));
    if(!Q.data)
        exit(OVERFLOW);
    Q.front = 0;
    Q.rear = 0;
    return OK;
}
Status EnQueue(SqQueue &Q, BiTree e)
{
    if ((Q.rear + 1) % MAXSIZE == Q.front) /* 队列满的判断 */
        return ERROR;
    Q.data[Q.rear] = e;              /* 将元素e赋值给队尾 */
    Q.rear = (Q.rear + 1) % MAXSIZE; /* rear指针向后移一位置， */
                                       /* 若到最后则转到数组头部 */
    return OK;
}
Status DeQueue(SqQueue &Q, BiTree &e)

{
    if (Q.front == Q.rear) /* 队列空的判断 */
        return ERROR;
    e = Q.data[Q.front];              /* 将队头元素赋值给e */
    Q.front = (Q.front + 1) % MAXSIZE; /* front指针向后移一位置， */
                                         /* 若到最后则转到数组头部 */
    return OK;
}

void LevelOrderTraverse(BiTree T)
{
    BiTree p;
    SqQueue qu;    //使用队列来层次遍历二叉树
    InitQueue(qu);
    EnQueue(qu, T);
    while(!QueueEmpty(qu))        //队列非空/*  */
    {
        DeQueue(qu, p);
        printf("%c", p->data);
        if(p->lchild)
            EnQueue(qu, p->lchild);
        if(p->rchild)
            EnQueue(qu, p->rchild);
    }
    return;
}
